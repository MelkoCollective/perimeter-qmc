98  void spin_update() {
99  grid_.init_loops();
100 
101  for(state_type bra = qmc::start_state; bra < qmc::n_bra; ++bra) {
102  grid_.alternator_ = bra;
103  std::for_each(grid_.begin(), grid_.end(),
104  [&](site_type & s) {
105  if(s.check[bra] == false)
106  {
107  if(rngS_() > .5) {
108  grid_.follow_loop_tpl(&s, bra,
109  [&](site_type * next) {
110  next->check[bra] = true;
111  }
112  );
113  }
114  else {
115  grid_.follow_loop_tpl(&s, bra,
116  [&](site_type * next) {
117  next->check[bra] = true;
118  next->spin[bra] = qmc::invert_spin - next->spin[bra];
119  }
120  );
121  #ifdef SIMUVIZ_FRAMES
122  simuviz_frame(1);
123  #endif //SIMUVIZ_FRAMES
124  }
125  }
126  }
127  );
128  }
129  grid_.clear_check();
130  }
134  void update() {
135  grid_.set_shift_mode(qmc::no_shift);
136 
137  for(state_type state = qmc::start_state; state < qmc::n_states; ++state)
138  for(unsigned i = 0; i < H_ * L_; ++i) {
139  bool ok = two_bond_update(rngH_(), rngL_(), state);
140  accept_ << ok;
141  #ifdef SIMUVIZ_FRAMES
142  if(ok)
143  simuviz_frame();
144  #endif //SIMUVIZ_FRAMES
145  }
146 
147  grid_.set_shift_mode(qmc::ket_preswap);
148  spin_update();
149  grid_.copy_to_ket(); //bc spins have changed
150  grid_.clear_tile_spin(); //bc spins have changed
151  }

158 
159  data_["loops"] << loops;
160  data_["overlap"] << pow(2.0, loops - 2*H_*L_* .5 );
161  //=================== swap zone ===================
162  grid_.set_shift_mode(qmc::ket_swap);
163  grid_.init_loops();
164  //grid_.eco_init_loops(); one could do a more efficient init_loop
165 
166  data_["sign"] << (grid_.sign() == 1);
167  data_["neg_loops"] << grid_.n_neg_loops() / (double)grid_.n_loops();
168  data_["swap_loops"] << grid_.n_loops();
169  data_["swap_overlap"] << pow(2.0, int(grid_.n_loops()) - loops);
170  data_["mean_for_error"] << pow(2.0, int(grid_.n_loops()) - loops);
171 
172  //=================== back to preswap ===================
173  grid_.set_shift_mode(qmc::ket_preswap);
174  }

220  //if -fix is found in the bash arguments it will just recalculate the jackknife
221  if(param_.find("fix") == param_.end()) {
222  //if -del is found in the bash arguments all progress/mean files will be deleted
223  if(param_.find("del") != param_.end()) {
224  addon::immortal.reset();
225  remove(mean_file.c_str());
226  remove((std::string(param_["prog_dir"]) + "/state.txt").c_str()); //timer...
227  }
228  if(addon::immortal.available()) { //else if there are progress files around they get loaded
229  std::cout << GREENB << "load data at index " << addon::immortal.get_index() << NONE << std::endl;
230  addon::immortal >> (*this);
231  }
232  else {//otherwise the thermalization begins normally
233  //------------------- therm -------------------
234  std::cout << std::endl;
235  for(unsigned i = 0; i < param_["term"]; ++i) {
236  update();
237  timer.progress(i, param_["timer_dest"]);
238  }
239  }
240  //------------------- sim -------------------
241  std::ofstream ofs;
242  for(unsigned i = addon::immortal.get_index(0); i < param_["sim"]; ++i) {
243 
244  update();
245  measure();
246  timer.progress(param_["term"] + i, param_["timer_dest"]);
247 
248  if((i & ((1lu<<10) - 1)) == ((1lu<<10) - 1)) { //all 1024 one mean gets added to the bins vector
249 
250  bins.push_back(data_["mean_for_error"].mean());
251  data_["mean_for_error"] = accumulator_double();
252 
253  if((i & ((1lu<<14) - 1)) == ((1lu<<14) - 1)) { //all 16*1024 the progress is saved and the bins written
254  //------------------- write out bins -------------------
255  write_bins(bins, mean_file);

258  addon::immortal.write_next_index(i + 1);
259  }
260  }
261  }
262  write_bins(bins, mean_file);
263  timer.write_state(param_["term"] + param_["sim"]);
264  }
265 
266

268 
269  timer.write(addon::global_seed.get()

281  }
