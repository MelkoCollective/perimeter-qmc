207  void init_loops() {
208  n_loops_ = 0;
209  n_neg_loops_ = 0;
210  sign_ = +1;
211  int subsign = +1;
212 
213  for(state_type bra = qmc::start_state; bra < qmc::n_bra; ++bra) //all transition graphs
214  std::for_each(begin(), end(), // all sites in a transition graph
215  [&](site_type & s) {
216  if(s.check[bra] == false) { //only if not already visited
217  subsign = +1;
218  alternator_ = bra; //must be bra, not ket, see subsign *= -1 below
219  auto old_bra = bra;
220  site_type::last_dir = 0;
221 
222  follow_loop_tpl(&s, bra,
223  [&](site_type * next){
224  next->check[bra] = true;
225  next->loop[bra] = n_loops_;
226  assert(alternator_ == bra or alternator_ == qmc::invert_state - bra);
227 
228  if(alternator_ == bra) {
229  if(site_type::last_dir >= qmc::middle) //will never happen at the beginning since last_dir == 0
230  subsign *= -1;
231  }
232  else {
233  if(site_type::last_dir < qmc::middle)
234  subsign *= -1;
235  }
236  }
237  );
238  if(site_type::last_dir >= qmc::middle) //here we need the bra case
239  subsign *= -1;
240 
241  assert(old_bra == bra); //we must end in the same layer we started
242 
243  if(subsign == -1) {
244  ++n_neg_loops_;
245  sign_ *= -1;
246  }
247 
248  ++n_loops_;
249  }
250  }
251  );
252  clear_check();
253  }

384  void init_grid(std::vector<unsigned> const init) {
385  int state = 0;
386 
387  for(state_type bra = qmc::start_state; bra < qmc::n_bra; ++bra) {
388  state = 0;
389  std::for_each(begin(), end(),
390  [&](site_type & s) {
391  state_type ket = qmc::invert_state - bra;
392  s.spin[bra] = (state + state / L_)%2 == 0 ? qmc::beta : qmc::alpha;
393  s.spin[ket] = s.spin[bra];
394 
395  if(init[bra] == 0) {
396  if(qmc::n_bonds == qmc::hex) {
397  s.bond[bra] = qmc::hori;
398  s.bond[ket] = qmc::hori;
399  }
400  else {
401  s.bond[bra] = (state%2==0 ? qmc::right:qmc::left);
402  s.bond[ket] = (state%2==0 ? qmc::right:qmc::left);
403  }
404  }
405  else if(init[bra] == 1) {
406  s.bond[bra] = (state/L_%2==0 ? qmc::down:qmc::up);
407  s.bond[ket] = (state/L_%2==0 ? qmc::down:qmc::up);
408  }
409  else if(init[bra] == 2) {
410  if(qmc::n_bonds == qmc::hex) {
411  s.bond[bra] = (state/L_%3==0 ? qmc::down: (state/L_%3==2 ? qmc::hori : qmc::up));
412  s.bond[ket] = (state/L_%3==0 ? qmc::down: (state/L_%3==2 ? qmc::hori : qmc::up));
413  }
414  else if(qmc::n_bonds == qmc::tri) {
415  s.bond[bra] = (state/L_%2==0 ? qmc::diag_down:qmc::diag_up);
416  s.bond[ket] = (state/L_%2==0 ? qmc::diag_down:qmc::diag_up);
417  }
418  }
419  s.loop[bra] = 1-(state + state / L_)%2; //important for tile_init hex
420  ++state;
421  }
422  );
423  }
424  //initialising the neighbor structure
425  for(unsigned i = 0; i < H_; ++i) {
426  for(unsigned j = 0; j < L_; ++j) {
427  grid_[i][j].neighbor[qmc::up] = &grid_[(i+H_-1)%H_][j];
428  grid_[i][j].neighbor[qmc::down] = &grid_[(i+1)%H_][j];
429  grid_[i][j].neighbor[qmc::me] = &grid_[i][j]; //me pointer points to opject itself (for monomers)
430 
431  //periodic boundaries happen here
432  if(qmc::n_bonds == qmc::hex) {
433  grid_[i][j].neighbor[qmc::hori] = &grid_[i][(j+L_ + 1 - 2*((i+j)%2) )%L_];
434  }
435  else {
436  grid_[i][j].neighbor[qmc::left] = &grid_[i][(j+L_-1)%L_];
437  grid_[i][j].neighbor[qmc::right] = &grid_[i][(j+1)%L_];
438  }
439 
440  if(qmc::n_bonds == qmc::tri) {
441  grid_[i][j].neighbor[qmc::diag_up] = &grid_[(i+H_-1)%H_][(j+L_-1)%L_];
442  grid_[i][j].neighbor[qmc::diag_down] = &grid_[(i+1)%H_][(j+1)%L_];
443  }
444  }
445  }
