54  site_struct * loop_partner(bond_type & state, bond_type & bra, shift_type const & shift_mode) {
55  if(shift_mode == qmc::no_shift) {
56  last_dir = bond[state];
57  return neighbor[bond[state]];
58  }
59 
60  if(state != bra) {//it's a ket
61  state_type new_ket = state - shift_region[shift_mode];
62 
63  if(new_ket < qmc::n_bra) //lazy boundary for now
64  new_ket += qmc::n_bra;
65 
66 
67  site_struct * partner = neighbor[bond[new_ket]];
68  last_dir = bond[new_ket];
69  if(partner->shift_region[shift_mode] != shift_region[shift_mode]) {
70  bra += qmc::n_bra - (partner->shift_region[shift_mode] - shift_region[shift_mode]);
71 
72  if(bra >= qmc::n_bra) //lazy boundary for now
73  bra %= qmc::n_bra;
74 
75  state = qmc::invert_state - bra;
76  }
77  return partner;
78  }
79  else {
80  last_dir = bond[state];
81  return neighbor[bond[state]];
82  }

197  std::stringstream res;
198  state_type s2 = qmc::invert_state - s1;

203  if(shift_region[shift_mode_print] != 0) {
204  ket_color = YELLOWB;
205  trans_color = WHITEB;
206  s2 -= shift_region[shift_mode_print];

210  }
211 
212
